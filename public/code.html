TradingView Indicator

Cross Trend Color (CTC) Strategy indicator for TradingView. Displays the Trend Magic line with color-coded buy/sell signals, HTF touch confirmation, and entry levels. 

//@version=6
indicator("CTC Strategy", overlay=true, max_labels_count=500)

//==============================
// TREND MAGIC SETTINGS
//==============================
period = input.int(15, "CCI Period")
coeff  = input.float(1.0, "ATR Multiplier")
AP     = input.int(5, "ATR Period")
src    = input.source(close, "Source")

// PER-SESSION TRADE TRACKING
maxTradesLondon = input.int(2, "Max Trades — London", minval=1, maxval=10, step=1)
maxTradesNewYork = input.int(2, "Max Trades — New York", minval=1, maxval=10, step=1)
var int londonTradeCount = 0
var int newyorkTradeCount = 0
var int utBotLondonTradeCount = 0
var int utBotNewYorkTradeCount = 0
var int lastDay = dayofmonth

if dayofmonth != lastDay
    londonTradeCount := 0
    newyorkTradeCount := 0
    utBotLondonTradeCount := 0
    utBotNewYorkTradeCount := 0
    lastDay := dayofmonth

//==============================
// TREND MAGIC
//==============================
ATR = ta.sma(ta.tr, AP)

upT   = low - ATR * coeff
downT = high + ATR * coeff

var float MagicTrend = na

MagicTrend := ta.cci(src, period) >= 0 ?
     (upT < nz(MagicTrend[1]) ? nz(MagicTrend[1]) : upT) :
     (downT > nz(MagicTrend[1]) ? nz(MagicTrend[1]) : downT)

trendBlue = ta.cci(src, period) >= 0
trendRed  = ta.cci(src, period) < 0

plot(
     MagicTrend,
     title="Trend Magic",
     color=trendBlue ? color.blue : color.red,
     linewidth=3
)

//==============================
// UT BOT SETTINGS
//==============================
showUtBot = input.bool(true, "Show UT Bot", group="UT Bot Settings")
utBotKey = input.float(1.0, "UT Bot Key Value", step=0.1, group="UT Bot Settings")
utBotAtrPeriod = input.int(10, "UT Bot ATR Period", minval=1, maxval=50, step=1, group="UT Bot Settings")
utBotHeikinAshi = input.bool(false, "UT Bot Heikin Ashi Candles", group="UT Bot Settings")
utBotMaxTradesLondon = input.int(2, "UT Bot Max Trades — London", minval=1, maxval=10, step=1, group="UT Bot Settings")
utBotMaxTradesNewYork = input.int(2, "UT Bot Max Trades — New York", minval=1, maxval=10, step=1, group="UT Bot Settings")
utBotBarColor = input.bool(true, "UT Bot Bar Coloring", group="UT Bot Settings")

//==============================
// FILTER SETTINGS
//==============================
usePrevCandleFilter = input.bool(false, "Use Previous Candle Filter", group="Filter Settings")

useHtfFilter = input.bool(false, "Use HTF Trend Filter", group="Filter Settings")
htfTimeframe = input.string("60", "HTF Timeframe", options=["15","30","60","120","240","D","W"], group="Filter Settings")

useHtfTouchFilter = input.bool(true, "Use HTF Touch Filter", group="Filter Settings")
htfTouchBars = input.int(3, "HTF Touch Bars", minval=1, maxval=10, step=1, group="Filter Settings")

//==============================
// SESSION SETTINGS
//==============================
londonSession = input.session("0300-0440", "London Session (EST)")
newyorkSession = input.session("0800-0955", "New York Session (EST)")

inLondonSession = not na(time(timeframe.period, londonSession, "America/New_York"))
inNewYorkSession = not na(time(timeframe.period, newyorkSession, "America/New_York"))

bgcolor(
     inLondonSession ?
     #00897b1a :
     na
)

bgcolor(
     inNewYorkSession ?
     #f236451a :
     na
)

//==============================
// UT BOT TRAILING STOP
//==============================
utBotXatr = ta.atr(utBotAtrPeriod)
utBotNLoss = utBotKey * utBotXatr

utBotSrc = utBotHeikinAshi ? request.security(ticker.heikinashi(syminfo.tickerid), timeframe.period, close, lookahead=barmerge.lookahead_off) : close

var float utBotTrailingStop = 0.0
utBotTrailingStop := utBotSrc > nz(utBotTrailingStop[1]) and utBotSrc[1] > nz(utBotTrailingStop[1]) ?
     math.max(nz(utBotTrailingStop[1]), utBotSrc - utBotNLoss) :
     utBotSrc < nz(utBotTrailingStop[1]) and utBotSrc[1] < nz(utBotTrailingStop[1]) ?
     math.min(nz(utBotTrailingStop[1]), utBotSrc + utBotNLoss) :
     utBotSrc > nz(utBotTrailingStop[1]) ? utBotSrc - utBotNLoss : utBotSrc + utBotNLoss

var int utBotPos = 0
utBotPos := utBotSrc[1] < nz(utBotTrailingStop[1]) and utBotSrc > nz(utBotTrailingStop[1]) ? 1 :
     utBotSrc[1] > nz(utBotTrailingStop[1]) and utBotSrc < nz(utBotTrailingStop[1]) ? -1 :
     nz(utBotPos[1], 0)

utBotStopColor = utBotPos == -1 ? color.red : utBotPos == 1 ? color.green : color.blue

utBotEma = ta.ema(utBotSrc, 1)
utBotAbove = ta.crossover(utBotEma, utBotTrailingStop)
utBotBelow = ta.crossover(utBotTrailingStop, utBotEma)

utBotBuyRaw = utBotSrc > utBotTrailingStop and utBotAbove
utBotSellRaw = utBotSrc < utBotTrailingStop and utBotBelow

// UT Bot session-restricted signals
utBotBuySignal = showUtBot and utBotBuyRaw and
     (inLondonSession and utBotLondonTradeCount < utBotMaxTradesLondon or inNewYorkSession and utBotNewYorkTradeCount < utBotMaxTradesNewYork)

utBotSellSignal = showUtBot and utBotSellRaw and
     (inLondonSession and utBotLondonTradeCount < utBotMaxTradesLondon or inNewYorkSession and utBotNewYorkTradeCount < utBotMaxTradesNewYork)

//==============================
// HTF TREND CONFIRMATION
//==============================
// Request HTF data (default lookahead so we can see complete HTF bars)
[htfCciRaw, htfCloseRaw] = request.security(syminfo.tickerid, htfTimeframe, [ta.cci(close, period), close])

// Detect when a new HTF bar starts (previous one just closed)
isNewHtfBar = timeframe.change(htfTimeframe)

// Store CONFIRMED values from the PREVIOUS completed HTF bar only
// This prevents the HTF line from moving with the current forming candle
var float htfClose = na
var float htfCci = na
if isNewHtfBar
    htfClose := htfCloseRaw[1]
    htfCci := htfCciRaw[1]

// Initialize on first bar
if na(htfClose)
    htfClose := htfCloseRaw
    htfCci := htfCciRaw

htfTrendBull = htfCci >= 0
htfTrendOk = not useHtfFilter or htfTrendBull == trendBlue

//==============================
// HTF TOUCH CONFIRMATION — default on
//==============================
// Check if a candle's range (high to low) touches the HTF line
candleTouchesHtf = high >= htfClose and low <= htfClose

// At least 1 of the recent N candles must have touched the HTF line
anyTouchHtf = false
for i = 0 to htfTouchBars - 1
    if candleTouchesHtf[i]
        anyTouchHtf := true

// Buy: close must be above HTF line
buyHtfTouchOk = not useHtfTouchFilter or (close > htfClose and anyTouchHtf)
// Sell: close must be below HTF line
sellHtfTouchOk = not useHtfTouchFilter or (close < htfClose and anyTouchHtf)

// HTF VISUAL OVERLAY
showHtfVisual = input.bool(true, "Show HTF Visual", group="Filter Settings")

// Plot HTF close as a stepped line (holds value constant between HTF bars)
htfPlotColor = htfTrendBull ? color.new(color.blue, 30) : color.new(color.red, 30)
plot(showHtfVisual ? htfClose : na, "HTF Line", htfPlotColor, 2, plot.style_stepline)

strongBuyCross =
     open < MagicTrend and
     close > MagicTrend

strongSellCross =
     open > MagicTrend and
     close < MagicTrend

//==============================
// BUY CONDITIONS
//==============================
buySignal =
     (inLondonSession and londonTradeCount < maxTradesLondon or inNewYorkSession and newyorkTradeCount < maxTradesNewYork) and
     strongBuyCross and
     trendBlue and
     htfTrendOk and
     buyHtfTouchOk

//==============================
// SELL CONDITIONS
//==============================
sellSignal =
     (inLondonSession and londonTradeCount < maxTradesLondon or inNewYorkSession and newyorkTradeCount < maxTradesNewYork) and
     strongSellCross and
     trendRed and
     htfTrendOk and
     sellHtfTouchOk

// PREVIOUS CANDLE CONFIRMATION — prev candle must already be positioned in direction (no cross needed)
prevBullish = close[1] > open[1]
prevBearish = close[1] < open[1]

// PREV CANDLE OPEN vs TREND LINE — prev candle open must be on correct side of MagicTrend
prevOpenBelowTrend = open[1] < MagicTrend[1]
prevOpenAboveTrend = open[1] > MagicTrend[1]

confirmedBuy = buySignal and (not usePrevCandleFilter or prevBullish) and prevOpenBelowTrend
confirmedSell = sellSignal and (not usePrevCandleFilter or prevBearish) and prevOpenAboveTrend

showInfoTable = input.bool(true, "Show Info Table")

//==============================
// ENTRY LEVEL & VALIDATION
//==============================
entryLevelPct = input.float(0.5, "Entry Level (%)", minval=0.0, maxval=1.0, step=0.05)
entryLevelLabel = str.tostring(entryLevelPct * 100, "#") + "%"

var line entryLine = na
var label entryLabel = na
var float pendingPrice = na
var int pendingBar = -1
var string pendingDir = ""

// Validate pending entry: must hit entry level within 1st or 2nd candle
if not na(pendingPrice) and not na(pendingBar)
    barsSince = bar_index - pendingBar
    hit = false

    // Check 1st and 2nd candle after entry candle
    if barsSince >= 1 and barsSince <= 2
        if pendingDir == "buy" and low <= pendingPrice
            hit := true
        if pendingDir == "sell" and high >= pendingPrice
            hit := true

    if hit
        // Entry validated — update to confirmed style
        if not na(entryLine)
            line.set_color(entryLine, pendingDir == "buy" ? color.lime : color.red)
            line.set_width(entryLine, 2)
        if not na(entryLabel)
            newText = pendingDir == "buy" ? "BUY ✓" : "SELL ✓"
            label.set_text(entryLabel, newText)
            label.set_color(entryLabel, pendingDir == "buy" ? color.green : color.red)
        pendingPrice := na
        pendingBar := -1
        pendingDir := ""

    if barsSince > 2 and not hit
        // Entry invalid — price never reached the level in time
        if not na(entryLine)
            line.set_color(entryLine, color.gray)
            line.set_width(entryLine, 1)
        if not na(entryLabel)
            newText2 = pendingDir == "buy" ? "BUY" : "SELL"
            label.set_text(entryLabel, newText2)
            label.set_color(entryLabel, color.gray)
        pendingPrice := na
        pendingBar := -1
        pendingDir := ""

// --- BUY ENTRY ---
if confirmedBuy
    entryPrice = low + (high - low) * entryLevelPct

    if not na(entryLine)
        line.delete(entryLine)
    if not na(entryLabel)
        label.delete(entryLabel)

    entryLine := line.new(

    pendingBar := bar_index
    pendingDir := "buy"

// --- SELL ENTRY ---
if confirmedSell
    entryPrice = low + (high - low) * entryLevelPct

    if not na(entryLine)
        line.delete(entryLine)
    if not na(entryLabel)
        label.delete(entryLabel)

    entryLine := line.new(
         bar_index,
         entryPrice,
         bar_index + 3,
         entryPrice,

         bar_index,
         high,
         "SELL",
         style=label.style_label_down,

calc.ts AssemblyScript calculation logic — CCI, ATR, Trend Magic, crossover, HTF touch & session checks 
//==============================
// TREND MAGIC — StrategyTune
// Ported from TradingView Pine Script
//==============================

// Incremental state: committed value from the last completed bar
let prevMagicTrend: f64 = 0.0;
let magicTrend: f64 = 0.0;

// HTF stepped close state — timestamp-based hour detection
let htfLineVal: f64 = 0.0;
let prevHtfLineVal: f64 = 0.0;
let prevUtcHour: i32 = -1;

function calc(ctx: Context): void {
  const period: i32 = ctx.inputs.period;
  const coeff: f64 = ctx.inputs.coeff;
  const ap: i32 = ctx.inputs.ap;

  // Need enough bars for both ATR (ap) and CCI (period)
  let neededBars: i32 = period;
  if (ap > neededBars) neededBars = ap;
  neededBars += 1;
  if (ctx.bars.count < neededBars) return;

  //==============================
  // SESSION BACKGROUNDS
  // Based on EST (America/New_York). Uses UTC-5 offset.
  // Session start/end times are configurable with minute precision.
  // Defaults: London 03:00-05:00 EST, New York 08:00-10:00 EST.
  //
  // Note: DST (EDT, UTC-4) shifts these by 1 hour. To handle DST
  // accurately you'd need date-based DST detection (2nd Sun Mar -
  // 1st Sun Nov). For simplicity a constant EST offset is used.
  //==============================
  const t: i32 = ctx.bar.timestamp;
  // Total minutes since midnight in EST (UTC-5)
  const utcMinutes: i32 = (t / 60) % (24 * 60);
  const estMinutes: i32 = (utcMinutes - 5 * 60 + 24 * 60) % (24 * 60);

  // Inputs are in HHMM format (e.g. 830 = 08:30). Parse hour & minute.
  const londonStartMin: i32 = (ctx.inputs.londonStart / 100) * 60 + (ctx.inputs.londonStart % 100);
  const londonEndMin: i32 = (ctx.inputs.londonEnd / 100) * 60 + (ctx.inputs.londonEnd % 100);
  const newyorkStartMin: i32 = (ctx.inputs.newyorkStart / 100) * 60 + (ctx.inputs.newyorkStart % 100);
  const newyorkEndMin: i32 = (ctx.inputs.newyorkEnd / 100) * 60 + (ctx.inputs.newyorkEnd % 100);

  const inLondonSession: bool = estMinutes >= londonStartMin && estMinutes < londonEndMin;
  const inNewYorkSession: bool = estMinutes >= newyorkStartMin && estMinutes < newyorkEndMin;

  // Full-height column at 2x the bar's high spans the entire chart
  // vertically, simulating bgcolor() with 90% transparency.
  // Using a value proportional to price avoids axis-distortion risk.
  const fullBg: f64 = ctx.bar.high + Math.abs(ctx.bar.high);
  ctx.plots.londonBg = inLondonSession ? fullBg : NaN;
  ctx.plots.newyorkBg = inNewYorkSession ? fullBg : NaN;

  //==============================
  // ATR — SMA of True Range
  //==============================
  let trSum: f64 = 0.0;
  for (let i: i32 = 0; i < ap; i++) {
    const h: f64 = ctx.bars.high(i);
    const l: f64 = ctx.bars.low(i);
    const pc: f64 = ctx.bars.close(i + 1);
    const tr: f64 = Math.max(h - l, Math.max(Math.abs(h - pc), Math.abs(l - pc)));
    trSum += tr;
  }
  const atr: f64 = trSum / ap;

  //==============================
  // CCI — Commodity Channel Index
  // using source (resolved to ctx.bars.close by the platform)
  //==============================
  // Pass 1: SMA of source
  let srcSum: f64 = 0.0;
  for (let i: i32 = 0; i < period; i++) {
    srcSum += ctx.bars.close(i);
  }
  const srcSma: f64 = srcSum / period;

  // Pass 2: Mean Deviation
  let mdSum: f64 = 0.0;
  for (let i: i32 = 0; i < period; i++) {
    mdSum += Math.abs(ctx.bars.close(i) - srcSma);
  }
  const md: f64 = mdSum / period;

  let cci: f64 = 0.0;
  if (md != 0.0) {
    cci = (ctx.bar.close - srcSma) / (0.015 * md);
  }

  //==============================
  // MAGIC TREND
  //==============================
  const upT: f64 = ctx.bar.low - atr * coeff;
  const downT: f64 = ctx.bar.high + atr * coeff;

  if (ctx.barIndex == 0) {
    // First bar — initialise with whichever side CCI is on
    magicTrend = cci >= 0.0 ? upT : downT;
  } else if (ctx.isNewBar) {
    // Commit previous bar's value, then advance
    prevMagicTrend = magicTrend;
    if (cci >= 0.0) {
      // Bullish — trend line only moves up
      magicTrend = upT < prevMagicTrend ? prevMagicTrend : upT;
    } else {
      // Bearish — trend line only moves down
      magicTrend = downT > prevMagicTrend ? prevMagicTrend : downT;
    }
  } else {
    // Live tick — recompute tentative value from committed state
    if (cci >= 0.0) {
      magicTrend = upT < prevMagicTrend ? prevMagicTrend : upT;
    } else {
      magicTrend = downT > prevMagicTrend ? prevMagicTrend : downT;
    }
  }

  //==============================
  // HTF 1H LINE — Close from previous hour, stepped
  // Simulates request.security(ticker, "60", close) by detecting
  // actual hourly boundaries from the bar timestamp.
  // On the first bar of each new UTC hour, the line updates to the
  // previous bar's close (= the just-completed hour's close).
  // Works automatically on any chart timeframe — no config needed.
  //==============================
  const utcHour: i32 = (ctx.bar.timestamp / 3600) % 24;
  if (prevUtcHour < 0) {
    // First bar — initialise
    htfLineVal = ctx.bar.close;
    prevHtfLineVal = ctx.bar.close;
    prevUtcHour = utcHour;
  } else if (utcHour != prevUtcHour) {
    // New hour — step line to the previous bar's close
    prevHtfLineVal = htfLineVal;
    htfLineVal = ctx.bars.close(1);
    prevUtcHour = utcHour;
  }

  //==============================
  // PLOT
  //==============================
  ctx.plots.magicTrend = magicTrend;
  ctx.plots.magicTrendColor = cci >= 0.0 ? 0 : 1;

  // HTF line — stepped close, colored by HTF direction
  const htfBull: bool = htfLineVal >= prevHtfLineVal;
  ctx.plots.htfLine = ctx.inputs.showHtfLine != 0 ? htfLineVal : NaN;
